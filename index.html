Loading...

No invoices yet.
'; return; } // Group by month const groups = {}; allInvoices.forEach(inv => { const d = new Date(inv.invoice_date); const key = `${d.getFullYear()}-${String(d.getMonth()+1).padStart(2,'0')}`; const label = d.toLocaleString('en-IN', { month: 'long', year: 'numeric' }); if (!groups[key]) groups[key] = { label, invoices: [], total: 0 }; groups[key].invoices.push(inv); groups[key].total += Number(inv.amount); }); const sortedKeys = Object.keys(groups).sort((a,b) => b.localeCompare(a)); list.innerHTML = sortedKeys.map((key, idx) => { const g = groups[key]; const isOpen = idx === 0; // first month open by default return `

${g.invoices.map(inv => `

₹${fmt(inv.amount)}
${inv.consignor_name}
${dispDate(inv.invoice_date)} · ${fmtQ(inv.quantity)} Kgs · ₹${fmt(inv.rate)}/Kg
`).join('')}
`; }).join(''); } catch(e) { list.innerHTML = '
