async function doLogin() { const pin = document.getElementById('pinInput').value.trim(); if (!pin) return; const btn = document.getElementById('pinBtn'); const msg = document.getElementById('pinConnecting'); const err = document.getElementById('pinError'); btn.disabled = true; btn.textContent = 'Connecting...'; msg.textContent = 'Connecting to server...'; msg.classList.add('show'); err.style.display = 'none'; try { const r = await fetch(`${API}/auth/login`,{ method:'POST', headers:{'Content-Type':'application/json'}, body:JSON.stringify({pin}) }); const d = await r.json(); if (d.success) { sessionStorage.setItem('urp_auth','1'); msg.textContent = 'Loading companies...'; const rc = await fetch(`${API}/companies`); allCompanies = await rc.json(); msg.classList.remove('show'); renderCompanyList(); showScreen('company'); } else { err.textContent = 'Incorrect PIN. Try again.'; err.style.display = 'block'; document.getElementById('pinInput').value = ''; } } catch(e) { err.textContent = 'Server starting up. Wait 30 seconds and try again.'; err.style.display = 'block'; } btn.disabled = false; btn.textContent = 'Unlock'; msg.classList.remove('show'); }
